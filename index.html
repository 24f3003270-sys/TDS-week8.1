Scenario Analysis (Code Sample)

Example: simple JavaScript function for CAGR calculation.


// Compute Compound Annual Growth Rate (CAGR)
function cagr(beginValue, endValue, years) {
  if (beginValue <= 0 || years <= 0) {
    throw new Error("Invalid inputs");
  }
  return Math.pow(endValue / beginValue, 1 / years) - 1;
}

// Example: Net income from 1200 to 1800 over 3 years
const result = cagr(1200, 1800, 3);
console.log("CAGR:", (result * 100).toFixed(2) + "%");
